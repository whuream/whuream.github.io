    public static <F, T> T reduce(final Iterator<F> iterator,
                                  final T begin,
                                  final Function<? super F, T> function) {
        T ret = begin;
        while (iterator.hasNext()) {
            ret = function.apply(iterator.next(), ret);
        }

        return ret;
    }

    public interface Function<F, T> {

        T apply(F f, T t);
    }
